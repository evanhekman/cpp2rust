#
# T*, int n          ->  &[T]            fold length param into slice

# ptr++ / ptr--      ->  index variable  pointer arithmetic -> index
# *ptr               ->  slice[idx]      dereference -> index
# throw / catch      ->  return          exception -> early return
# a[i] * b[i] (int)  ->  (a[i] as u32) * (b[i] as u32)   type widening before op
# int return         ->  u32 or i32      semantic type selection
# for (p=a+2; p<a+n; p+=2)  ->  while i < a.len() { ... i += 2; }   stride loop
